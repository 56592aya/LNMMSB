5 function add_aliases(sym::Symbol, aliases::Symbol...)
6 for alias in aliases
7 if haskey(_keyAliases, alias)
8 error("Already an alias $alias => $(_keyAliases[alias])... can't also alias $sym")
9 end
10 _keyAliases[alias] = sym
11 end
12 end

983 groupIds::Vector{Vector{Int}} # list of indices for each group

987 # this is when given a vector-type of values to group by
988 function extractGroupArgs(v::AVec, args...; legendEntry = string)
989 groupLabels = sort(collect(unique(v)))
990 n = length(groupLabels)
991 if n > 100
992 warn("You created n=$n groups... Is that intended?")
993 end
994 groupIds = Vector{Int}[filter(i -> v[i] == glab, 1:length(v)) for glab in groupLabels]
995 GroupBy(map(legendEntry, groupLabels), groupIds)
996 end

998 legendEntryFromTuple(ns::Tuple) = string(("$n " for n in ns)...)
999
1000 # this is when given a tuple of vectors of values to group by
1001 function extractGroupArgs(vs::Tuple, args...)
1002 (vs == ()) && return GroupBy([""], [1:size(args[1],1)])
1003 v = collect(zip(vs...))
1004 extractGroupArgs(v, args...; legendEntry = legendEntryFromTuple)
1005 end

1009 groupLabels = sortedkeys(idxmap)
1010 groupIds = Vector{Int}[collect(idxmap[k]) for k in groupLabels]
1011 GroupBy(groupLabels, groupIds)
