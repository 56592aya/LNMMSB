155 function optimal_ticks_and_labels(axis::Axis, ticks = nothing)
156 amin,amax = axis_limits(axis)
157
158 # scale the limits
159 scale = axis[:scale]
160 sf = scalefunc(scale)
161
162 # If the axis input was a Date or DateTime use a special logic to find
163 # "round" Date(Time)s as ticks
164 # This bypasses the rest of optimal_ticks_and_labels, because
165 # optimize_datetime_ticks returns ticks AND labels: the label format (Date
166 # or DateTime) is chosen based on the time span between amin and amax
167 # rather than on the input format
168 # TODO: maybe: non-trivial scale (:ln, :log2, :log10) for date/datetime
169 if ticks == nothing && scale == :identity
170 if axis[:formatter] == dateformatter
171 # optimize_datetime_ticks returns ticks and labels(!) based on
172 # integers/floats corresponding to the DateTime type. Thus, the axes
173 # limits, which resulted from converting the Date type to integers,
174 # are converted to 'DateTime integers' (actually floats) before
175 # being passed to optimize_datetime_ticks.
176 # (convert(Int, convert(DateTime, convert(Date, i))) == 87600000*i)
177 ticks, labels = optimize_datetime_ticks(864e5 * amin, 864e5 * amax;
178 k_min = 2, k_max = 4)
179 # Now the ticks are converted back to floats corresponding to Dates.
180 return ticks / 864e5, labels
181 elseif axis[:formatter] == datetimeformatter
182 return optimize_datetime_ticks(amin, amax; k_min = 2, k_max = 4)
183 end
184 end
185
186 # get a list of well-laid-out ticks
187 if ticks == nothing
188 scaled_ticks = optimize_ticks(
189 sf(amin),
190 sf(amax);
191 k_min = 4, # minimum number of ticks
192 k_max = 8, # maximum number of ticks
193 )[1]
194 elseif typeof(ticks) <: Int
195 scaled_ticks, viewmin, viewmax = optimize_ticks(
196 sf(amin),
197 sf(amax);
198 k_min = ticks, # minimum number of ticks
199 k_max = ticks, # maximum number of ticks
200 k_ideal = ticks,
201 # `strict_span = false` rewards cases where the span of the
202 # chosen ticks is not too much bigger than amin - amax:
203 strict_span = false,
204 )
205 axis[:lims] = map(invscalefunc(scale), (viewmin, viewmax))
206 else
207 scaled_ticks = map(sf, (filter(t -> amin <= t <= amax, ticks)))
208 end
209 unscaled_ticks = map(invscalefunc(scale), scaled_ticks)
210
211 labels = if any(isfinite, unscaled_ticks)
212 formatter = axis[:formatter]
213 if formatter == :auto
214 # the default behavior is to make strings of the scaled values and then apply the labelfunc
215 map(labelfunc(scale, backend()), Showoff.showoff(scaled_ticks, :plain))
216 elseif formatter == :scientific
217 Showoff.showoff(unscaled_ticks, :scientific)
218 else
219 # there was an override for the formatter... use that on the unscaled ticks
220 map(formatter, unscaled_ticks)
221 end
222 else
223 # no finite ticks to show...
224 String[]
225 end
226
227 # @show unscaled_ticks labels
228 # labels = Showoff.showoff(unscaled_ticks, scale == :log10 ? :scientific : :auto)
229 unscaled_ticks, labels
