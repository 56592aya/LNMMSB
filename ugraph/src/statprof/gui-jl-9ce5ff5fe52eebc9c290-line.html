33 # get or set the default GUI; doesn't affect running GUI
34 """
35 pygui()
36
37 Return the current GUI toolkit as a symbol.
38 """
39 function pygui()
40 global gui
41 if gui::Symbol != :default && pygui_works(gui::Symbol)
42 return gui::Symbol
43 else
44 for g in (:tk, :qt, :wx, :gtk, :gtk3)
45 if pygui_works(g)
46 gui = g
47 return gui::Symbol
48 end
49 end
50 error("No supported Python GUI toolkit is installed.")
51 end
52 end
53 function pygui(g::Symbol)
54 global gui
55 if g != gui::Symbol
56 if !(g in (:wx,:gtk,:gtk3,:tk,:qt,:qt4,:qt5,:qt_pyqt5,:qt_pyqt4,:qt_pyside,:default))
57 throw(ArgumentError("invalid gui $g"))
58 elseif !pygui_works(g)
59 error("Python GUI toolkit for $g is not installed.")
60 end
61 gui = g
62 end
63 return g::Symbol
64 end

86 function gtk_eventloop(gtkmodule::AbstractString, sec::Real=50e-3)
87 gtk_requireversion(gtkmodule)

99 # As discussed in PyPlot.jl#278, Qt looks for a file qt.conf in
100 # the same path as the running executable, which tells it where
101 # to find plugins etcetera. For Python's Qt, however, this will
102 # be in the path of the python executable, not julia. Furthermore,
103 # we can't copy it to the location of julia (even if that location
104 # is writable) because that would assume that all julia programs
105 # use the same version of Qt (e.g. via the same Python), which
106 # is not necessarily the case, and even then it wouldn't work
107 # for other programs linking libjulia. Unfortunately, there
108 # seems to be no way to change this. However, we can at least
109 # use set QT_PLUGIN_PATH by parsing qt.conf ourselves, and
110 # this seems to fix some of the path-related problems on Windows.
111 # ... unfortunately, it seems fixqtpath has to be called before
112 # the Qt library is loaded.
113 function fixqtpath(qtconf=joinpath(dirname(pyprogramname),"qt.conf"))
114 haskey(ENV, "QT_PLUGIN_PATH") && return false
115 if isfile(qtconf)
116 for line in eachline(qtconf)
117 m = match(r"^\s*prefix\s*=(.*)$"i, line)
118 if m !== nothing
119 dir = strip(m.captures[1])
120 if startswith(dir, '"') && endswith(dir, '"')
121 dir = dir[2:end-1]
122 end
123 plugin_path = joinpath(dir, "plugins")
124 if isdir(plugin_path)
125 # for some reason I don't understand,
126 # if libpython has already been loaded, then
127 # we need to use Python's setenv rather than Julia's
128 PyDict(pyimport("os")["environ"])["QT_PLUGIN_PATH"] = realpath(plugin_path)
129 return true
130 end
131 end
132 end
133 end
134 return false
135 end

137 # Qt: (PyQt5, PyQt4, or PySide module)
138 function qt_eventloop(QtCore::PyObject, sec::Real=50e-3)
139 fixqtpath()
140 instance = QtCore["QCoreApplication"]["instance"]
141 AllEvents = QtCore["QEventLoop"]["AllEvents"]
142 processEvents = QtCore["QCoreApplication"]["processEvents"]
143 pop!(ENV, "QT_PLUGIN_PATH", "") # clean up environment
144 maxtime = PyObject(50)
145 install_doevent(sec) do async
146 app = pycall(instance, PyObject)
147 if app.o != pynothing[]
148 app["_in_event_loop"] = true
149 pycall(processEvents, PyObject, AllEvents, maxtime)
150 end
151 end
152 end

166 # wx: (based on IPython/lib/inputhookwx.py, which is 3-clause BSD-licensed)
167 function wx_eventloop(sec::Real=50e-3)
168 wx = pyimport("wx")
169 GetApp = wx["GetApp"]
170 EventLoop = wx["EventLoop"]
171 EventLoopActivator = wx["EventLoopActivator"]
172 install_doevent(sec) do async
173 app = pycall(GetApp, PyObject)
174 if app.o != pynothing[]
175 app["_in_event_loop"] = true
176 evtloop = pycall(EventLoop, PyObject)
177 ea = pycall(EventLoopActivator, PyObject, evtloop)
178 Pending = evtloop["Pending"]
179 Dispatch = evtloop["Dispatch"]
180 while pycall(Pending, Bool)
181 pycall(Dispatch, PyObject)
182 end
183 pydecref(ea) # deactivate event loop
184 pycall(app["ProcessIdle"], PyObject)
185 end
186 end
187 end

199 # cache running event loops (so that we don't start any more than once)
200 const eventloops = Dict{Symbol,Timer}()
201
202 """
203 pygui_start(gui::Symbol = pygui())
204
205 Start the event loop of a certain toolkit.
206
207 The argument `gui` defaults to the current default GUI, but it could be `:wx`, `:gtk`, `:gtk3`, `:tk`, or `:qt`.
208
209 """
210 function pygui_start(gui::Symbol=pygui(), sec::Real=50e-3)
211 pygui(gui)
212 if !haskey(eventloops, gui)
213 if gui == :wx
214 eventloops[gui] = wx_eventloop(sec)
215 elseif gui == :gtk
216 eventloops[gui] = gtk_eventloop("gtk", sec)
217 elseif gui == :gtk3
218 eventloops[gui] = gtk_eventloop("gi.repository.Gtk", sec)
219 elseif gui == :tk
220 eventloops[gui] = Tk_eventloop(sec)
221 elseif gui == :qt_pyqt5 || gui == :qt5
222 eventloops[gui] = qt_eventloop("PyQt5", sec)
223 elseif gui == :qt_pyqt4
224 eventloops[gui] = qt_eventloop("PyQt4", sec)
225 elseif gui == :qt_pyside
226 eventloops[gui] = qt_eventloop("PySide", sec)
227 elseif gui == :qt4
228 try
229 eventloops[gui] = qt_eventloop("PyQt4", sec)
230 catch
231 eventloops[gui] = qt_eventloop("PySide", sec)
232 end
233 elseif gui == :qt
234 eventloops[gui] = qt_eventloop(sec)
235 else
236 throw(ArgumentError("unsupported GUI type $gui"))
237 end
238 end
239 gui
240 end

243 pygui_stop(gui::Symbol = pygui())
244
245 Stop any running event loop for gui. The `gui` argument defaults to current default GUI.

248 function pygui_stop(gui::Symbol=pygui())
249 if haskey(eventloops, gui)

254 end
