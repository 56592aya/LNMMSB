95 function create_kw_body(func_signature::Expr)
96 # get the arg list, stripping out any keyword parameters into a
97 # bunch of get!(kw, key, value) lines
98 func_signature.head == :where && return create_kw_body(func_signature.args[1])
99 args = func_signature.args[2:end]
100 kw_body = Expr(:block)
101 cleanup_body = Expr(:block)
102 if isa(args[1], Expr) && args[1].head == :parameters
103 for kwpair in args[1].args
104 k, v = kwpair.args
105 push!(kw_body.args, :($k = get!($PLOTATTRIBUTES, $(QuoteNode(k)), $v)))
106 push!(cleanup_body.args, :(RecipesBase.is_key_supported($(QuoteNode(k))) || delete!($PLOTATTRIBUTES, $(QuoteNode(k)))))
107 end
108 args = args[2:end]
109 end
110 args, kw_body, cleanup_body
111 end

113 # process the body of the recipe recursively.
114 # when we see the series macro, we split that block off:
115 # let
116 # d2 = copy(d)
117 # <process_recipe_body on d2>
118 # RecipeData(d2, block_return)
119 # end
120 # and we push this block onto the series_blocks list.
121 # then at the end we push the main body onto the series list
122 function process_recipe_body!(expr::Expr)
123 for (i,e) in enumerate(expr.args)
124 if isa(e,Expr)
125
126 # process trailing flags, like:
127 # a --> b, :quiet, :force
128 quiet, require, force = false, false, false
129 if _is_arrow_tuple(e)
130 for flag in e.args
131 if _equals_symbol(flag, :quiet)
132 quiet = true
133 elseif _equals_symbol(flag, :require)
134 require = true
135 elseif _equals_symbol(flag, :force)
136 force = true
137 end
138 end
139 e = e.args[1]
140 end
141
142 # the unused operator `:=` will mean force: `x := 5` is equivalent to `x --> 5, force`
143 # note: this means "x is defined as 5"
144 if e.head == :(:=)
145 force = true
146 e.head = :(-->)
147 end
148
149 # we are going to recursively swap out `a --> b, flags...` commands
150 # note: this means "x may become 5"
151 if e.head == :(-->)
152 k, v = e.args
153 if isa(k, Symbol)
154 k = QuoteNode(k)
155 end
156
157 set_expr = if force
158 # forced override user settings
159 :($PLOTATTRIBUTES[$k] = $v)
160 else
161 # if the user has set this keyword, use theirs
162 :(get!($PLOTATTRIBUTES, $k, $v))
163 end
164
165 expr.args[i] = if quiet
166 # quietly ignore keywords which are not supported
167 :(RecipesBase.is_key_supported($k) ? $set_expr : nothing)
168 elseif require
169 # error when not supported by the backend
170 :(RecipesBase.is_key_supported($k) ? $set_expr : error("In recipe: required keyword ", $k, " is not supported by backend $(backend_name())"))
171 else
172 set_expr
173 end
174
175 # TODO elseif it's a @series macrocall, add a series block and push to the `series` list
176
177 elseif e.head != :call
178 # we want to recursively replace the arrows, but not inside function calls
179 # as this might include things like Dict(1=>2)
180 process_recipe_body!(e)
181 end
182 end
183 end
184 end

196 push!(new_args, new_arg)
197 end
198 Expr(ex.head, new_args...), n

202 """
203 This handy macro will process a function definition, replace `-->` commands, and
204 then add a new version of `RecipesBase.apply_recipe` for dispatching on the arguments.
205
206 This functionality is primarily geared to turning user types and settings into the
207 data and attributes that describe a Plots.jl visualization.
208
209 Set attributes using the `-->` command, and return a comma separated list of arguments that
210 should replace the current arguments.
211
212 An example:
213
214 ```
215 using RecipesBase
216
217 # Our custom type that we want to display
218 type T end
219
220 @recipe function plot{N<:Integer}(t::T, n::N = 1; customcolor = :green)
221 markershape --> :auto, :require
222 markercolor --> customcolor, :force
223 xrotation --> 5
224 zrotation --> 6, :quiet
225 rand(10,n)
226 end
227
228 # ---------------------
229
230 # Plots will be the ultimate consumer of our recipe in this example
231 using Plots; gr()
232
233 # This call will implicitly call `RecipesBase.apply_recipe` as part of the Plots
234 # processing pipeline (see the Pipeline section of the Plots documentation).
235 # It will plot 5 line plots, all with black circles for markers.
236 # The markershape argument must be supported, and the zrotation argument's warning
237 # will be suppressed. The user can override all arguments except markercolor.
238 plot(T(), 5; customcolor = :black, shape=:c)
239 ```
240
241 In this example, we see lots of the machinery in action. We create a new type `T` which
242 we will use for dispatch, and an optional argument `n`, which will be used to determine the
243 number of series to display. User-defined keyword arguments are passed through, and the
244 `-->` command can be trailed by flags:
245
246 - quiet: Suppress unsupported keyword warnings
247 - require: Error if keyword is unsupported
248 - force: Don't allow user override for this keyword
249 """
250 macro recipe(funcexpr::Expr)
251 funcexpr, n_depr = _replace(funcexpr, PLOTATTRIBUTES_OLD, PLOTATTRIBUTES)
252 func_signature, func_body = funcexpr.args
253
254 if n_depr > 0
255 msg = "Usage of `$PLOTATTRIBUTES_OLD` for accessing plot attributes is deprecated. Found $n_depr usages of `$PLOTATTRIBUTES_OLD` inside `@recipe`. Use `$PLOTATTRIBUTES` instead.\n"
256 deprecation_body = :(Base.depwarn($msg, :d))
257 else
258 deprecation_body = :()
259 end
260 if !(funcexpr.head in (:(=), :function))
261 error("Must wrap a valid function call!")
262 end
263 if !(isa(func_signature, Expr) && func_signature.head in (:call, :where))
264 error("Expected `func_signature = ...` with func_signature as a call or where Expr... got: $func_signature")
265 end
266 if length(func_signature.args) < 2
267 error("Missing function arguments... need something to dispatch on!")
268 end
269
270 args, kw_body, cleanup_body = create_kw_body(func_signature)
271 func = get_function_def(func_signature, args)
272
273 # this is where the receipe func_body is processed
274 # replace all the key => value lines with argument setting logic
275 # and break up by series.
276 process_recipe_body!(func_body)
277
278 # now build a function definition for apply_recipe, wrapping the return value in a tuple if needed.
279 # we are creating a vector of RecipeData objects, one per series.
280 funcdef = Expr(:function, func, esc(quote
281 if RecipesBase._debug_recipes[1]
282 println("apply_recipe args: ", $args)
283 end
284 $kw_body
285 $cleanup_body
286 series_list = RecipesBase.RecipeData[]
287 func_return = $func_body
288 if func_return != nothing
289 push!(series_list, RecipesBase.RecipeData($PLOTATTRIBUTES, RecipesBase.wrap_tuple(func_return)))
290 end
291 series_list
292 end))
293 funcdef
294 end
