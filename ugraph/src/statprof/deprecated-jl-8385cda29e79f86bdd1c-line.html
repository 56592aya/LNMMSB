75 function _depwarn(msg, opts, bt, caller)
76 ln = Int(unsafe_load(cglobal(:jl_lineno, Cint)))
77 fn = unsafe_string(unsafe_load(cglobal(:jl_filename, Ptr{Cchar})))
78 if opts.depwarn == 1 # raise a warning
79 warn(msg, once=(caller != StackTraces.UNKNOWN), key=(caller,fn,ln), bt=bt,
80 filename=fn, lineno=ln)
81 elseif opts.depwarn == 2 # raise an error
82 throw(ErrorException(msg))
83 end

87 function firstcaller(bt::Array{Ptr{Void},1}, funcsyms)
88 # Identify the calling line
89 found = false
90 lkup = StackTraces.UNKNOWN
91 for frame in bt
92 lkups = StackTraces.lookup(frame)
93 for lkup in lkups
94 if lkup == StackTraces.UNKNOWN
95 continue
96 end
97 found && @goto found
98 found = lkup.func in funcsyms
99 # look for constructor type name
100 if !found && !isnull(lkup.linfo)
101 li = get(lkup.linfo)
102 ft = ccall(:jl_first_argument_datatype, Any, (Any,), li.def.sig)
103 if isa(ft,DataType) && ft.name === Type.body.name
104 ft = unwrap_unionall(ft.parameters[1])
105 found = (isa(ft,DataType) && ft.name.name in funcsyms)
106 end
107 end
108 end
109 end
110 return StackTraces.UNKNOWN
111 @label found
112 return lkup
113 end

908 function produce(v)
909 depwarn("produce is now deprecated. Use Channels for inter-task communication.", :produce)
910
911 ct = current_task()
912 local empty, t, q
913 while true
914 q = ct.consumers
915 if isa(q,Task)
916 t = q
917 ct.consumers = nothing
918 empty = true
919 break
920 elseif isa(q,Condition) && !isempty(q.waitq)
921 t = shift!(q.waitq)
922 empty = isempty(q.waitq)
923 break
924 end
925 wait()
926 end
927
928 t.state == :runnable || throw(AssertionError("producer.consumer.state == :runnable"))
929 if empty
930 schedule_and_wait(t, v)
931 while true
932 # wait until there are more consumers
933 q = ct.consumers
934 if isa(q,Task)
935 return q.result
936 elseif isa(q,Condition) && !isempty(q.waitq)
937 return q.waitq[1].result
938 end
939 wait()
940 end
941 else
942 schedule(t, v)
943 # make sure `t` runs before us. otherwise, the producer might
944 # finish before `t` runs again, causing it to see the producer
945 # as done, causing done(::Task, _) to miss the value `v`.
946 # see issue #7727
947 yield()
948 return q.waitq[1].result
949 end
