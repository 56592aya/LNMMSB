286 """
287 pywrap(o::PyObject)
288
289 This returns a wrapper `w` that is an anonymous module which provides (read) access to converted versions of o's members as w.member.
290
291 For example, `@pyimport module as name` is equivalent to `const name = pywrap(pyimport("module"))`
292
293 If the Python module contains identifiers that are reserved words in Julia (e.g. function), they cannot be accessed as `w.member`; one must instead use `w.pymember(:member)` (for the PyAny conversion) or w.pymember("member") (for the raw PyObject).
294 """
295 function pywrap(o::PyObject, mname::Symbol=:__anon__)
296 members = convert(Vector{Tuple{AbstractString,PyObject}},
297 pycall(inspect["getmembers"], PyObject, o))
298 filter!(m -> !(m[1] in reserved), members)
299 m = Module(mname, false)
300 consts = [Expr(:const, Expr(:(=), Symbol(x[1]), convert(PyAny, x[2]))) for x in members]
301 exports = try
302 convert(Vector{Symbol}, o["__all__"])
303 catch
304 [Symbol(x[1]) for x in filter(x -> x[1][1] != '_', members)]
305 end
306 eval(m, Expr(:toplevel, consts..., :(pymember(s) = $(getindex)($(o), s)),
307 Expr(:export, exports...)))
308 m
309 end

657 kw = PyObject(Dict{AbstractString, Any}([Pair(string(k), v) for (k, v) in kwargs]))
658 ret = PyObject(@pycheckn ccall((@pysym :PyObject_Call), PyPtr,
659 (PyPtr,PyPtr,PyPtr), o, arg, kw))
660 end
661 return ret::PyObject
662 finally
663 sigatomic_end()
664 end
665 end

669
670 Call the given Python function (typically looked up from a module) with the given args... 682 """
683 @pycall func(args...)::T
684
685 Convenience macro which turns `func(args...)::T` into pycall(func, T, args...)
686 """
687 macro pycall(ex)
688 if !(isexpr(ex,:(::)) && isexpr(ex.args[1],:call))
689 throw(ArgumentError("Usage: @pycall func(args...)::T"))
690 end
691 func = ex.args[1].args[1]
692 args, kwargs = ex.args[1].args[2:end], []
693 if isexpr(args[1],:parameters)
694 kwargs, args = args[1], args[2:end]
695 end
696 T = ex.args[2]
697 :(pycall($(map(esc,[kwargs; func; T; args])...)))
698 end
