552 expct_col = false
553 state = 4
554 elseif !is_cr
555 state = 2
556 expct_col = false
557 end
558 elseif 3 == state # second quote
559 if is_quote && !was_cr
560 state = 1
561 elseif is_dlm && !was_cr
562 state = 0
563 col += 1
564 store_cell(dh, nrows+1, col, true, col_start_idx, idx-2)
565 col_start_idx = idx
566 !ign_adj_dlm && (expct_col = true)
567 elseif is_eol
568 nrows += 1
569 col += 1
570 store_cell(dh, nrows, col, true, col_start_idx, idx - (was_cr ? 3 : 2))
571 col_start_idx = idx
572 ncols = max(ncols, col)
573 col = 0
574 state = 0
575 elseif is_comment && allow_comments && !was_cr
576 nrows += 1
577 col += 1
578 store_cell(dh, nrows, col, true, col_start_idx, idx - 2)
579 ncols = max(ncols, col)
580 col = 0
581 state = 4
582 elseif (is_cr && was_cr) || !is_cr
583 error_str = escape_string("unexpected character '$(Char(val))' after quoted field at row $(nrows+1) column $(col+1)")
584 break
585 end
586 elseif 5 == state # skip start
587 if is_eol
588 col_start_idx = idx
589 skipstart -= 1
590 (0 == skipstart) && (state = 0)
591 end
592 end
593 was_cr = is_cr
594 end
595
596 if isempty(error_str)
597 if 1 == state # quoted field
598 error_str = "truncated column at row $(nrows+1) column $(col+1)"
599 elseif (2 == state) || (3 == state) || ((0 == state) && is_dlm) # unquoted field, second quote, or begin field with last character as delimiter
600 col += 1
601 nrows += 1
602 store_cell(dh, nrows, col, (3 == state), col_start_idx, idx-1)
603 ncols = max(ncols, col)
604 end
605 end
606 catch ex
607 if isa(ex, TypeError) && (ex.func == :store_cell)
608 rethrow(ex)
609 else
610 error("at row $(nrows+1), column $col : $ex)")
611 end
612 end
613 !isempty(error_str) && error(error_str)
614
615 return (nrows, ncols)
616 end

623 function writedlm_cell{T}(io::IO, elt::AbstractString, dlm::T, quotes::Bool)
624 if quotes && !isempty(elt) && (('"' in elt) || ('\n' in elt) || ((T <: Char) ? (dlm in elt) : contains(elt, dlm)))
625 print(io, '"', replace(elt, r"\"", "\"\""), '"')
626 else
627 print(io, elt)
628 end
629 end
630 writedlm_cell(io::IO, elt, dlm, quotes) = print(io, elt)
631 function writedlm(io::IO, a::AbstractMatrix, dlm; opts...)
632 optsd = val_opts(opts)
633 quotes = get(optsd, :quotes, true)
634 pb = PipeBuffer()
635 lastc = last(indices(a, 2))
636 for i = indices(a, 1)
637 for j = indices(a, 2)
638 writedlm_cell(pb, a[i, j], dlm, quotes)
639 j == lastc ? write(pb,'\n') : print(pb,dlm)
640 end
641 (nb_available(pb) > (16*1024)) && write(io, take!(pb))
642 end
643 write(io, take!(pb))
644 nothing
645 end

649 # write an iterable row as dlm-separated items
650 function writedlm_row(io::IO, row, dlm, quotes)
651 state = start(row)
652 while !done(row, state)
653 (x, state) = next(row, state)
654 writedlm_cell(io, x, dlm, quotes)
655 done(row, state) ? write(io,'\n') : print(io,dlm)
656 end
657 end

659 # If the row is a single string, write it as a string rather than
660 # iterating over characters. Also, include the common case of
661 # a Number (handled correctly by the generic writedlm_row above)
662 # purely as an optimization.
663 function writedlm_row(io::IO, row::Union{Number,AbstractString}, dlm, quotes)
664 writedlm_cell(io, row, dlm, quotes)
665 write(io, '\n')
666 end

687 """
688 writedlm(f, A, delim='\\t'; opts)
689
690 Write `A` (a vector, matrix, or an iterable collection of iterable rows) as text to `f`
691 (either a filename string or an `IO` stream) using the given delimiter
692 `delim` (which defaults to tab, but can be any printable Julia object, typically a `Char` or
693 `AbstractString`).
694
695 For example, two vectors `x` and `y` of the same length can be written as two columns of
696 tab-delimited text to `f` by either `writedlm(f, [x y])` or by `writedlm(f, zip(x, y))`.
697 """
698 writedlm(io, a; opts...) = writedlm(io, a, '\t'; opts...)
