38 function _wishart_c0(df::Real, S::AbstractPDMat)
39 h_df = df / 2
40 p = dim(S)

83 function meanlogdet(d::Wishart)
84 p = dim(d)
85 df = d.df
86 v = logdet(d.S) + p * logtwo
87 for i = 1:p
88 v += digamma(0.5 * (df - (i - 1)))
89 end
90 return v

93 function entropy(d::Wishart)
94 p = dim(d)
95 df = d.df
96 d.c0 - 0.5 * (df - p - 1) * meanlogdet(d) + 0.5 * df * p

116 function _wishart_genA(p::Int, df::Real)
117 # Generate the matrix A in the Bartlett decomposition
118 #
119 # A is a lower triangular matrix, with
120 #
121 # A(i, j) ~ sqrt of Chisq(df - i + 1) when i == j
122 # ~ Normal() when i > j
123 #
124 A = zeros(p, p)
125 for i = 1:p
126 @inbounds A[i,i] = sqrt(rand(Chisq(df - i + 1.0)))
127 end
128 for j = 1:p-1, i = j+1:p
129 @inbounds A[i,j] = randn()
130 end
131 return A
