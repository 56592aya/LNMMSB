1789 function logm(A0::UpperTriangular{T}) where T<:Union{Float64,Complex{Float64}}
1790 maxsqrt = 100
1791 theta = [1.586970738772063e-005,
1792 2.313807884242979e-003,
1793 1.938179313533253e-002,
1794 6.209171588994762e-002,
1795 1.276404810806775e-001,
1796 2.060962623452836e-001,
1797 2.879093714241194e-001]
1798 tmax = size(theta, 1)
1799 n = size(A0, 1)
1800 A = copy(A0)
1801 p = 0
1802 m = 0
1803
1804 # Compute repeated roots
1805 d = complex(diag(A))
1806 dm1 = d .- 1
1807 s = 0
1808 while norm(dm1, Inf) > theta[tmax] && s < maxsqrt
1809 d .= sqrt.(d)
1810 dm1 .= d .- 1
1811 s = s + 1
1812 end
1813 s0 = s
1814 for k = 1:min(s, maxsqrt)
1815 A = sqrtm(A)
1816 end
1817
1818 AmI = A - I
1819 d2 = sqrt(norm(AmI^2, 1))
1820 d3 = cbrt(norm(AmI^3, 1))
1821 alpha2 = max(d2, d3)
1822 foundm = false
1823 if alpha2 <= theta[2]
1824 m = alpha2 <= theta[1] ? 1 : 2
1825 foundm = true
1826 end
1827
1828 while !foundm
1829 more = false
1830 if s > s0
1831 d3 = cbrt(norm(AmI^3, 1))
1832 end
1833 d4 = norm(AmI^4, 1)^(1/4)
1834 alpha3 = max(d3, d4)
1835 if alpha3 <= theta[tmax]
1836 for j = 3:tmax
1837 if alpha3 <= theta[j]
1838 break
1839 end
1840 end
1841 if j <= 6
1842 m = j
1843 break
1844 elseif alpha3 / 2 <= theta[5] && p < 2
1845 more = true
1846 p = p + 1
1847 end
1848 end
1849
1850 if !more
1851 d5 = norm(AmI^5, 1)^(1/5)
1852 alpha4 = max(d4, d5)
1853 eta = min(alpha3, alpha4)
1854 if eta <= theta[tmax]
1855 j = 0
1856 for j = 6:tmax
1857 if eta <= theta[j]
1858 m = j
1859 break
1860 end
1861 end
1862 break
1863 end
1864 end
1865
1866 if s == maxsqrt
1867 m = tmax
1868 break
1869 end
1870 A = sqrtm(A)
1871 AmI = A - I
1872 s = s + 1
1873 end
1874
1875 # Compute accurate superdiagonal of T
1876 p = 1 / 2^s
1877 for k = 1:n-1
1878 Ak = A0[k,k]
1879 Akp1 = A0[k+1,k+1]
1880 Akp = Ak^p
1881 Akp1p = Akp1^p
1882 A[k,k] = Akp
1883 A[k+1,k+1] = Akp1p
1884 if Ak == Akp1
1885 A[k,k+1] = p * A0[k,k+1] * Ak^(p-1)
1886 elseif 2 * abs(Ak) < abs(Akp1) || 2 * abs(Akp1) < abs(Ak)
1887 A[k,k+1] = A0[k,k+1] * (Akp1p - Akp) / (Akp1 - Ak)
1888 else
1889 logAk = log(Ak)
1890 logAkp1 = log(Akp1)
1891 w = atanh((Akp1 - Ak)/(Akp1 + Ak)) + im*pi*ceil((imag(logAkp1-logAk)-pi)/(2*pi))
1892 dd = 2 * exp(p*(logAk+logAkp1)/2) * sinh(p*w) / (Akp1 - Ak)
1893 A[k,k+1] = A0[k,k+1] * dd
1894 end
1895 end
1896
1897 # Compute accurate diagonal of T
1898 for i = 1:n
1899 a = A0[i,i]
1900 if s == 0
1901 r = a - 1
1902 end
1903 s0 = s
1904 if angle(a) >= pi / 2
1905 a = sqrt(a)
1906 s0 = s - 1
1907 end
1908 z0 = a - 1
1909 a = sqrt(a)
1910 r = 1 + a
1911 for j = 1:s0-1
1912 a = sqrt(a)
1913 r = r * (1 + a)
1914 end
1915 A[i,i] = z0 / r
1916 end
1917
1918 # Get the Gauss-Legendre quadrature points and weights
1919 R = zeros(Float64, m, m)
1920 for i = 1:m - 1
1921 R[i,i+1] = i / sqrt((2 * i)^2 - 1)
1922 R[i+1,i] = R[i,i+1]
1923 end
1924 x,V = eig(R)
1925 w = Vector{Float64}(m)
1926 for i = 1:m
1927 x[i] = (x[i] + 1) / 2
1928 w[i] = V[1,i]^2
1929 end
1930
1931 # Compute the Padé approximation
1932 Y = zeros(T, n, n)
1933 for k = 1:m
1934 Y = Y + w[k] * (A / (x[k] * A + I))
1935 end
1936
1937 # Scale back
1938 scale!(2^s, Y)
1939
1940 # Compute accurate diagonal and superdiagonal of log(T)
1941 for k = 1:n-1
1942 Ak = A0[k,k]
1943 Akp1 = A0[k+1,k+1]
1944 logAk = log(Ak)
1945 logAkp1 = log(Akp1)
1946 Y[k,k] = logAk
1947 Y[k+1,k+1] = logAkp1
1948 if Ak == Akp1
1949 Y[k,k+1] = A0[k,k+1] / Ak
1950 elseif 2 * abs(Ak) < abs(Akp1) || 2 * abs(Akp1) < abs(Ak)
1951 Y[k,k+1] = A0[k,k+1] * (logAkp1 - logAk) / (Akp1 - Ak)
1952 else
1953 w = atanh((Akp1 - Ak)/(Akp1 + Ak) + im*pi*(ceil((imag(logAkp1-logAk) - pi)/(2*pi))))
1954 Y[k,k+1] = 2 * A0[k,k+1] * w / (Akp1 - Ak)
1955 end
1956 end
1957
1958 return UpperTriangular(Y)
