37 # Recursion + Mutable State = Job Security
38 function dirsnearby(dir; descend = 1, ascend = 1, set = Set{String}())
39 push!(set, dir)
40 if descend > 0
41 for down in dirs(dir)
42 if !(down in set)
43 push!(set, down)
44 descend > 1 && dirsnearby(down, descend = descend-1, ascend = 0, set = set)
45 end
46 end
47 end
48 ascend > 0 && dirsnearby(dirname(dir), descend = descend, ascend = ascend-1, set = set)
49 return set

function codemodule(code, line)
76 stack = String[]
77 # count all unterminated block openers, brackets, and parens
78 n_openers = 0
79 n_brackets = 0
80 n_parens = 0
81 # index of next modulename token
82 next_modulename = -1
83
84 ts = tokenize(code)
85
86 last_token = nothing
87
88 1 (0.0023%)
1 (100.00%) samples spent calling next
for (i, t) in enumerate(ts)
89 Tokens.startpos(t)[1] > line && break
90
91 # Ignore everything in brackets or parnetheses, because any scope started in
92 # them also needs to be closed in them. That way, we don't need special
93 # handling for comprehensions and `end`-indexing.
94 if Tokens.kind(t) == Tokens.LSQUARE
95 n_brackets += 1
96 elseif n_brackets > 0
97 if Tokens.kind(t) == Tokens.RSQUARE
98 n_brackets -= 1
99 end
100 elseif Tokens.kind(t) == Tokens.LPAREN
101 n_parens += 1
102 elseif n_parens > 0
103 if Tokens.kind(t) == Tokens.RPAREN
104 n_parens -= 1
105 end
106 elseif Tokens.exactkind(t) in MODULE_STARTERS && (last_token == nothing || Tokens.kind(last_token) == Tokens.WHITESPACE) # new module
107 next_modulename = i + 2
108 elseif i == next_modulename && Tokens.kind(t) == Tokens.IDENTIFIER && Tokens.kind(last_token) == Tokens.WHITESPACE
109 push!(stack, Tokens.untokenize(t))
110 elseif Tokens.exactkind(t) in SCOPE_STARTERS # new non-module scope
111 n_openers += 1
112 elseif Tokens.exactkind(t) == Tokens.END # scope ended
113 n_openers == 0 ? (!isempty(stack) && pop!(stack)) : n_openers -= 1
114 end
115 last_token = t
116 end
117
118 return join(stack, ".")
119 end

127 function includeline(file::AbstractString, included_file::AbstractString)
128 i = 0
129 open(file) do io
130 for (index, line) in enumerate(eachline(io))
131 m = match(r"include\(\"([a-zA-Z_\.\\/]*)\"\)", line)
132 if m != nothing && normpath(joinpath(dirname(file), m.captures[1])) == included_file
133 i = index
134 break
135 end
136 end
137 end
138 return i

153 Takes an absolute path to a file and returns a string
154 representing the module it belongs to.
155 """
156 function filemodule_(path::AbstractString)
157 loc = find_include(path)
158 if loc != nothing
159 file, line = loc
160 mod = codemodule(readstring(file), line)
161 super = filemodule(file)
162 if super != "" && mod != ""
163 return "$super.$mod"
164 else

166 end
