4 """
5 ReentrantLock()
6
7 Creates a reentrant lock for synchronizing Tasks.
8 The same task can acquire the lock as many times as required.
9 Each `lock` must be matched with an `unlock`.
10
11 This lock is NOT threadsafe. See `Threads.Mutex` for a threadsafe lock.
12 """
13 mutable struct ReentrantLock
14 locked_by::Nullable{Task}
15 cond_wait::Condition
16 reentrancy_cnt::Int
17
18 ReentrantLock() = new(nothing, Condition(), 0)
19 end

21 """
22 islocked(the_lock) -> Status (Boolean)
23
24 Check whether the lock is held by any task/thread.
25 This should not be used for synchronization (see instead `trylock`).
26 """
27 function islocked(rl::ReentrantLock)
28 return rl.reentrancy_cnt != 0

31 """
32 trylock(the_lock) -> Success (Boolean)
33
34 Acquires the lock if it is available,
35 returning `true` if successful.
36 If the lock is already locked by a different task/thread,
37 returns `false`.
38
39 Each successful `trylock` must be matched by an `unlock`.
40 """
41 function trylock(rl::ReentrantLock)
42 t = current_task()
43 if rl.reentrancy_cnt == 0
44 rl.locked_by = t
45 rl.reentrancy_cnt = 1
46 return true
47 elseif t == get(rl.locked_by)
48 rl.reentrancy_cnt += 1
49 return true
50 end
51 return false
52 end

78 """
79 unlock(the_lock)
80
81 Releases ownership of the lock.
82
83 If this is a recursive lock which has been acquired before, it
84 just decrements an internal counter and returns immediately.
85 """
86 function unlock(rl::ReentrantLock)
87 if rl.reentrancy_cnt == 0
88 error("unlock count must match lock count")
89 end
90 rl.reentrancy_cnt -= 1
91 if rl.reentrancy_cnt == 0
92 rl.locked_by = nothing
93 notify(rl.cond_wait)
94 end
95 return
96 end

118 """
119 Semaphore(sem_size)
120
121 Creates a counting semaphore that allows at most `sem_size`
122 acquires to be in use at any time.
123 Each acquire must be mached with a release.
124
125 This construct is NOT threadsafe.
126 """
127 mutable struct Semaphore
128 sem_size::Int
129 curr_cnt::Int
130 cond_wait::Condition
131 Semaphore(sem_size) = sem_size > 0 ? new(sem_size, 0, Condition()) : throw(ArgumentError("Semaphore size must be > 0"))
132 end
133
134 """
135 acquire(s::Semaphore)
136
137 Wait for one of the `sem_size` permits to be available,
138 blocking until one can be acquired.
139 """
140 function acquire(s::Semaphore)
141 while true
142 if s.curr_cnt < s.sem_size
143 s.curr_cnt = s.curr_cnt + 1
144 return
145 else
146 wait(s.cond_wait)
147 end
148 end
149 end
150
151 """
152 release(s::Semaphore)
153
154 Return one permit to the pool,
155 possibly allowing another task to acquire it
156 and resume execution.
157 """
158 function release(s::Semaphore)
159 @assert s.curr_cnt > 0 "release count must match acquire count"
160 s.curr_cnt -= 1
161 notify(s.cond_wait; all=false)
162 end
