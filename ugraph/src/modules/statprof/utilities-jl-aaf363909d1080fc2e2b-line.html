129 function readutf(io, offset = 0)
130 ch = convert(UInt8, io.data[io.ptr + offset])
131 if ch < 0x80
132 return convert(Char, ch), 0
133 end
134 trailing = Base.utf8_trailing[ch + 1]
135 c::UInt32 = 0
136 for j = 1:trailing
137 c += ch
138 c <<= 6
139 ch = convert(UInt8, io.data[io.ptr + j + offset])
140 end
141 c += ch
142 c -= Base.utf8_offset[trailing + 1]
143 return convert(Char, c), trailing
144 end
