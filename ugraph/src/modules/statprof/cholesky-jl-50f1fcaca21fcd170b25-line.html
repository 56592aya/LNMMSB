7 # The dispatch structure in the chol!, chol, cholfact, and cholfact! methods is a bit
8 # complicated and some explanation is therefore provided in the following
9 #
10 # In the methods below, LAPACK is called when possible, i.e. StridedMatrices with Float32,
11 # Float64, Complex{Float32}, and Complex{Float64} element types. For other element or
12 # matrix types, the unblocked Julia implementation in _chol! is used. For cholfact
13 # and cholfact! pivoting is supported through a Val{Bool} argument. A type argument is
14 # necessary for type stability since the output of cholfact and cholfact! is either
15 # Cholesky or PivotedCholesky. The latter is only
16 # supported for the four LAPACK element types. For other types, e.g. BigFloats Val{true} will
17 # give an error. It is required that the input is Hermitian (including real symmetric) either
18 # through the Hermitian and Symmetric views or exact symmetric or Hermitian elements which
19 # is checked for and an error is thrown if the check fails. The dispatch
20 # is further complicated by a limitation in the formulation of Unions. The relevant union
21 # would be Union{Symmetric{T<:Real,S}, Hermitian} but, right now, it doesn't work in Julia
22 # so we'll have to define methods for the two elements of the union separately.
23
24 # FixMe? The dispatch below seems overly complicated. One simplification could be to
25 # merge the two Cholesky types into one. It would remove the need for Val completely but
26 # the cost would be extra unnecessary/unused fields for the unpivoted Cholesky and runtime
27 # checks of those fields before calls to LAPACK to check which version of the Cholesky
28 # factorization the type represents.

62 ## Non BLAS/LAPACK element types (generic)
63 function _chol!(A::AbstractMatrix, ::Type{UpperTriangular})
64 n = checksquare(A)
65 @inbounds begin
66 for k = 1:n
67 for i = 1:k - 1
68 A[k,k] -= A[i,k]'A[i,k]
69 end
70 Akk = _chol!(A[k,k], UpperTriangular)
71 A[k,k] = Akk
72 AkkInv = inv(Akk')
73 for j = k + 1:n
74 for i = 1:k - 1
75 A[k,j] -= A[i,k]'A[i,j]
76 end
77 A[k,j] = AkkInv*A[k,j]
78 end
79 end
80 end
81 return UpperTriangular(A)
82 end
83 function _chol!(A::AbstractMatrix, ::Type{LowerTriangular})
84 n = checksquare(A)
85 @inbounds begin
86 for k = 1:n
87 for i = 1:k - 1
88 A[k,k] -= A[k,i]*A[k,i]'
89 end
90 Akk = _chol!(A[k,k], LowerTriangular)
91 A[k,k] = Akk
92 AkkInv = inv(Akk)
93 for j = 1:k
94 for i = k + 1:n
95 if j == 1
96 A[i,k] = A[i,k]*AkkInv'
97 end
98 if j < k
99 A[i,k] -= A[i,j]*A[k,j]'*AkkInv'
100 end
101 end
102 end
103 end
104 end
105 return LowerTriangular(A)
106 end

558 `CC = cholfact(C[:U]'C[:U] + v*v')` but the computation of `CC` only uses `O(n^2)`
559 operations. The input factorization `C` is updated in place such that on exit `C == CC`.
560 The vector `v` is destroyed during the computation.
561 """
562 function lowrankupdate!(C::Cholesky, v::StridedVector)
563 A = C.factors
564 n = length(v)
565 if size(C, 1) != n
566 throw(DimensionMismatch("updating vector must fit size of factorization"))
567 end
568 if C.uplo == 'U'
569 conj!(v)
570 end
571
572 for i = 1:n
573
574 # Compute Givens rotation
575 c, s, r = givensAlgorithm(A[i,i], v[i])
576
577 # Store new diagonal element
578 A[i,i] = r
579
580 # Update remaining elements in row/column
581 if C.uplo == 'U'
582 for j = i + 1:n
583 Aij = A[i,j]
584 vj = v[j]
585 A[i,j] = c*Aij + s*vj
586 v[j] = -s'*Aij + c*vj
587 end
588 else
589 for j = i + 1:n
590 Aji = A[j,i]
591 vj = v[j]
592 A[j,i] = c*Aji + s*vj
593 v[j] = -s'*Aji + c*vj
594 end
595 end
596 end
597 return C
598 end

600 """
601 lowrankdowndate!(C::Cholesky, v::StridedVector) -> CC::Cholesky
602
603 Downdate a Cholesky factorization `C` with the vector `v`. If `A = C[:U]'C[:U]` then
604 `CC = cholfact(C[:U]'C[:U] - v*v')` but the computation of `CC` only uses `O(n^2)`
605 operations. The input factorization `C` is updated in place such that on exit `C == CC`.
606 The vector `v` is destroyed during the computation.
607 """
608 function lowrankdowndate!(C::Cholesky, v::StridedVector)
609 A = C.factors
610 n = length(v)
611 if size(C, 1) != n
612 throw(DimensionMismatch("updating vector must fit size of factorization"))
613 end
614 if C.uplo == 'U'
615 conj!(v)
616 end
617
618 for i = 1:n
619
620 Aii = A[i,i]
621
622 # Compute Givens rotation
623 s = conj(v[i]/Aii)
624 s2 = abs2(s)
625 if s2 > 1
626 throw(LinAlg.PosDefException(i))
627 end
628 c = sqrt(1 - abs2(s))
629
630 # Store new diagonal element
631 A[i,i] = c*Aii
632
633 # Update remaining elements in row/column
634 if C.uplo == 'U'
635 for j = i + 1:n
636 vj = v[j]
637 Aij = (A[i,j] - s*vj)/c
638 A[i,j] = Aij
639 v[j] = -s'*Aij + c*vj
640 end
641 else
642 for j = i + 1:n
643 vj = v[j]
644 Aji = (A[j,i] - s*vj)/c
645 A[j,i] = Aji
646 v[j] = -s'*Aji + c*vj
647 end
648 end
649 end
650 return C
651 end
